In [11]:
# make data for using k timestep
def make_data(train_data, test_data, k):
    x_train = []
    y_train = []
    x_test = []
    y_test = []
    
    # train data
    for i in range(k, train_data.shape[0]):
        x_train.append(train_data[i-k:i])
        y_train.append(train_data.iloc[i, 0])
    x_train, y_train = np.array(x_train), np.array(y_train)
    x_train = np.reshape(x_train, (x_train.shape[0], x_train.shape[1], 15))
    
    # test data
    for i in range(k, test_data.shape[0]):
        x_test.append(test_data[i-k:i])
        y_test.append(test_data.iloc[i, 0])
    x_test, y_test = np.array(x_test), np.array(y_test).reshape(-1, 1)
    x_test = np.reshape(x_test, (x_test.shape[0], x_test.shape[1], 15))
    
    return x_train, y_train, x_test, y_test

def run_rnn(train_data, test_data, k, l):
    # k for timestep, l for number of epochs

    # make data
    x_train, y_train, x_test, y_test = make_data(train_data, test_data, k)
    
    # init rnn
    model = Sequential()
    
    # layers
    model.add(LSTM(units = 60, return_sequences = True, input_shape = (x_train.shape[1], x_train.shape[2])))
    model.add(Dropout(0.3))
    model.add(LSTM(units = 60, return_sequences = True))
    model.add(Dropout(0.3))
    model.add(LSTM(units = 60, return_sequences = True))
    model.add(Dropout(0.3))
    model.add(LSTM(units = 60))
    model.add(Dropout(0.3))
    
    # output layer
    model.add(Dense(units = 1))
    
    # compile
    model.compile(optimizer = 'adam', loss = 'mean_squared_error')

    # fit to train set
    history = model.fit(x_train, y_train, epochs = l, batch_size = 64,
                        validation_data = (x_test, y_test))
    
    # predict
    predict = model.predict(x_test)
    predict = sd.inverse_transform(predict)

    y_test = sd.inverse_transform(y_test)

    return model, y_test, predict, history
